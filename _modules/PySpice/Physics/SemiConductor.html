        self.Is = Is # reverse bias saturation current
        self.n = n # ideality factor or emission coefficient
        self.T = Cst.temperature(degree=degree, kelvin=kelvin)

        return self.Is*(np.exp(Vd/(self.n*self.Vt)) - 1)
